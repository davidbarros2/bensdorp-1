    def test_guard_fires_when_db_exists(tmp_path: Path) -> None:
      Pre-create the DB file: (tmp_path / "data" / "bensdorp1.db") with .mkdir(parents=True) and .touch().

      Assert: result.exit_code == 1
      Assert: "Delete the file" in result.output (first recovery path)
      Assert: "bensdorp1 restore" in result.output (second recovery path)
      No input= needed (guard fires before any prompt).

  TEST 2 — Happy path completes successfully:
    def test_happy_path(tmp_path: Path) -> None:
      Patch all external calls (DATA_DIR, get_constituents, update_price_data, run_migrations, get_engine, create_backup, log_event).
      get_constituents return_value: {"AAPL": "Apple Inc.", "MSFT": "Microsoft Corporation"}
      get_engine return_value: a MagicMock() that supports context manager for engine.connect() (use MagicMock() — SQLAlchemy's engine mock is auto-specced by MagicMock for basic connect/execute/commit calls)
      Input: "y\n50000\ny\n" (Continue: y, cash: 50000, Confirm: y)
      Assert: result.exit_code == 0
      Assert: "Setup complete" in result.output
      Assert: "50,000.00" in result.output (formatted cash in completion summary)
      Assert log_event was called once (verify mock.called is True)
      Assert create_backup was called once

  TEST 3 — Cash validation re-prompts on invalid values:
    def test_cash_validation_reprompts(tmp_path: Path) -> None:
      Patch DATA_DIR, get_constituents (return_value={"AAPL": "Apple Inc."}), update_price_data, run_migrations, get_engine, create_backup, log_event.
      Input: "y\n0\n-100\n50000\ny\n"
        Explanation: Continue=y, cash=0 (invalid, re-prompt), cash=-100 (invalid, re-prompt), cash=50000 (valid), Confirm=y.
        Note: "abc" is handled by number_prompt() internally; do NOT include "abc" in the input string for this test because number_prompt re-prompts on non-numeric WITHOUT consuming a newline from our input. Test only the > 0 guard layer.
      Assert: result.exit_code == 0  (eventually succeeds)
      Assert: "Error: Cash must be greater than zero." in result.output  (appeared at least once)

  TEST 4 — Ctrl+C during cash entry aborts cleanly:
    def test_ctrl_c_during_cash_entry(tmp_path: Path) -> None:
      Patch DATA_DIR to tmp_path.
      Confirm prompt for "Continue?" returns True (simulate "y" so we reach the cash block).
      Then raise KeyboardInterrupt when number_prompt is called.

      Assert: "bensdorp1.db" not in str(tmp_path / "data" / "bensdorp1.db")  # DB file not created (no run_migrations called)
